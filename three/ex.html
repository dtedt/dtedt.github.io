// Smooth orientation controller class DeviceOrientationController { constructor() { this.gamma = 0; // Left/Right this.beta = 0; // Front/Back this.smoothing = 0.3; this.sensitivity = 0.12; this.deadZone = 3; this.enabled = false; } async init() { // iOS permission if (typeof DeviceOrientationEvent !== 'undefined' && typeof DeviceOrientationEvent.requestPermission === 'function') { document.getElementById('instructions').innerHTML = 'Tap to enable motion controls'; return new Promise((resolve) => { document.body.addEventListener('click', async () => { try { const permission = await DeviceOrientationEvent.requestPermission(); if (permission === 'granted') { window.addEventListener('deviceorientation', (e) => this.onOrientation(e)); this.enabled = true; resolve(true); } } catch (error) { console.error('Permission denied:', error); resolve(false); } }, { once: true }); }); } else { // Non-iOS devices window.addEventListener('deviceorientation', (e) => this.onOrientation(e)); this.enabled = true; return true; } } onOrientation(event) { // Apply dead zone first let rawGamma = Math.abs(event.gamma) < this.deadZone ? 0 : event.gamma; let rawBeta = Math.abs(event.beta) < this.deadZone ? 0 : event.beta; // Apply smoothing this.gamma = this.gamma * (1 - this.smoothing) + (rawGamma || 0) * this.smoothing; this.beta = this.beta * (1 - this.smoothing) + (rawBeta || 0) * this.smoothing; } getMovement() { return { x: this.gamma * this.sensitivity, y: this.beta * -this.sensitivity // Negative because beta is inverted }; } } // Use it in your game const orientation = new DeviceOrientationController(); async function init() { // ... your existing init code ... await orientation.init(); if (orientation.enabled) { document.getElementById('instructions').textContent = 'Tilt your device to navigate the dream'; } // ... rest of init ... } function updatePlayerPosition() { const movement = orientation.getMovement(); // Clamp to boundaries player.position.x = THREE.MathUtils.clamp( player.position.x + movement.x, -boundaries, boundaries ); player.position.y = THREE.MathUtils.clamp( player.position.y + movement.y, -boundaries, boundaries ); }
